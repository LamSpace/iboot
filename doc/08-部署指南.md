#### 8.2.2 各层部署配置

**接口层（Interfaces Layer）部署：**
- Controller 类无需特殊配置，Spring Boot 自动扫描
- 确保 `@RestController`、`@RequestMapping` 注解正确
- Swagger 文档在生产环境可禁用

**应用层（Application Layer）部署：**
- Application Service 使用 `@Service` 注解
- 事务边界定义在应用层（`@Transactional`）
- 确保领域事件发布器正确配置

**领域层（Domain Layer）部署：**
- 领域层是纯 Java 代码，无需特殊部署配置
- 确保领域模型、值对象、仓储接口正确定义
- 领域事件监听器使用 `@EventListener`

**基础设施层（Infrastructure Layer）部署：**
- Repository 实现类使用 `@Repository` 注解
- MyBatis Mapper 使用 `@Mapper` 或 XML 配置
- 确保数据源、Redis、MinIO 等连接配置正确

## Docker 容器化部署

### 8.10 Docker Compose 配置

**docker-compose.yml 完整配置：**

```yaml
version: '3.8'

services:
  # MySQL 数据库
  mysql:
    image: mysql:8.0
    container_name: iboot-mysql
    environment:
      MYSQL_ROOT_PASSWORD: root123
      MYSQL_DATABASE: iboot
      MYSQL_USER: iboot

      - ./data/mysql:/var/lib/mysql

      - --character-set-server=utf8mb4
      - --collation-server=utf8mb4_unicode_ci
      - --default-time-zone=+00:00

      - iboot-network
    restart: always

  # Redis 缓存
  redis:
    image: redis:7-alpine
    container_name: iboot-redis
    ports:
      - "6379:6379"
    volumes:
      - ./data/redis:/data

    networks:
      - iboot-network
    restart: always

  # MinIO 对象存储
  minio:
    image: minio/minio:latest
    container_name: iboot-minio
    environment:
      MINIO_ROOT_USER: minioadmin
      MINIO_ROOT_PASSWORD: minioadmin123
    ports:
      - "9000:9000"
      - "9001:9001"
    volumes:
      - ./data/minio:/data
    command: server /data --console-address ":9001"
    networks:
      - iboot-network
    restart: always

  # 后端服务
  iboot-admin:
    build:
      context: ../iboot-admin
      dockerfile: Dockerfile
    container_name: iboot-admin
    environment:
      DB_HOST: mysql
      DB_PORT: 3306
      DB_NAME: iboot
      DB_USERNAME: iboot

      REDIS_PORT: 6379

      - "8080:8080"
    depends_on:
      - mysql
      - redis
    networks:
      - iboot-network
    restart: always

  # 前端服务 (Nginx)
  iboot-portal:
    build:
      context: ../iboot-portal
      dockerfile: Dockerfile
    container_name: iboot-portal

      - "80:80"
    depends_on:
      - iboot-admin
    networks:
      - iboot-network
    restart: always

    image: prom/prometheus:latest
    container_name: iboot-prometheus

      - "9090:9090"

      - ./prometheus.yml:/etc/prometheus/prometheus.yml
      - ./data/prometheus:/prometheus
    networks:
      - iboot-network
    restart: always

  # Grafana 可视化

    image: grafana/grafana:latest
    container_name: iboot-grafana
    environment:
      GF_SECURITY_ADMIN_USER: admin
      GF_SECURITY_ADMIN_PASSWORD: admin123
    ports:
      - "3000:3000"
    volumes:
      - ./data/grafana:/var/lib/grafana
      - ./grafana/dashboards:/etc/grafana/provisioning/dashboards
      - ./grafana/datasources:/etc/grafana/provisioning/datasources
    depends_on:
      - prometheus
    networks:
      - iboot-network
    restart: always

networks:
  iboot-network:
    driver: bridge
```

```nginx
upstream backend {
    least_conn;
    server 192.168.1.100:8080 weight=3;

    keepalive 32;

server {
    listen 80;
    server_name your-domain.com;

    # HTTP 重定向到 HTTPS
    return 301 https://$server_name$request_uri;
}

server {
    listen 443 ssl http2;
    server_name your-domain.com;

    # SSL 证书配置
    ssl_certificate /etc/nginx/ssl/your-domain.crt;
    ssl_certificate_key /etc/nginx/ssl/your-domain.key;
    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers HIGH:!aNULL:!MD5;
    ssl_prefer_server_ciphers on;

    # 前端静态文件
    location / {
        root /var/www/iboot-portal;
        index index.html;
        try_files $uri $uri/ /index.html;

        # 浏览器缓存
        location ~* \.(jpg|jpeg|png|gif|ico|css|js|svg|woff|woff2|ttf|eot)$ {
            expires 1y;
            add_header Cache-Control "public, immutable";
        }
    }

    # API 反向代理
    location /api/ {
        proxy_pass http://backend/;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
        proxy_connect_timeout 60s;
        proxy_send_timeout 60s;
        proxy_read_timeout 60s;
    }

    # 访问日志
    access_log /var/log/nginx/iboot-access.log;
    error_log /var/log/nginx/iboot-error.log;

    # 开启 Gzip
    gzip on;
    gzip_vary on;
    gzip_min_length 1024;
    gzip_proxied any;
    gzip_comp_level 6;
    gzip_types text/plain text/css text/xml text/javascript application/json application/javascript application/xml+rss application/rss+xml font/truetype font/opentype application/vnd.ms-fontobject image/svg+xml;
}
```

## CI/CD 部署

### 8.18 GitHub Actions 配置

**.github/workflows/deploy.yml：**

```yaml
name: Deploy

on:
  push:
    branches: [ main ]
  pull_request:
    branches: [ main ]

jobs:
  build-and-deploy:
    runs-on: ubuntu-latest

    steps:
    - uses: actions/checkout@v3

    - name: Set up JDK 17
      uses: actions/setup-java@v3
      with:
        java-version: '17'
        distribution: 'temurin'

    - name: Set up Node.js 20
      uses: actions/setup-node@v3
      with:
        node-version: '20'

    - name: Build Backend
      run: |
        cd iboot-admin
        mvn clean package -DskipTests

    - name: Build Frontend
      run: |
        cd iboot-portal
        npm install
        npm run build

    - name: Deploy to Server
      uses: easingthemes/ssh-deploy@v3
      with:
        SSH_PRIVATE_KEY: ${{ secrets.SSH_KEY }}
        REMOTE_HOST: ${{ secrets.SERVER_HOST }}
        REMOTE_USER: ${{ secrets.SERVER_USER }}
        SOURCE: "iboot-admin/target/*.jar"
        TARGET: "/opt/iboot/"

    - name: Restart Service
      uses: appleboy/ssh-action@master
      with:
```

### 8.19 Jenkins Pipeline

**Jenkinsfile：**

```groovy
pipeline {
    agent any

    environment {
        DOCKER_REGISTRY = 'your-registry.com'
        IMAGE_VERSION = "${BUILD_NUMBER}"
    }

    stages {
        stage('Checkout') {
            steps {
                git branch: 'main', url: 'https://github.com/your-repo/iboot.git'
            }
        }

        stage('Build Backend') {
            steps {
                dir('iboot-admin') {
                    sh 'mvn clean package -DskipTests'
                }
            }
        }

        stage('Build Frontend') {
            steps {
                dir('iboot-portal') {
                    sh 'npm install'
                    sh 'npm run build'
                }
            }
        }

        stage('Build Docker Image') {
            steps {
                sh "docker build -f iboot-admin/Dockerfile -t ${DOCKER_REGISTRY}/iboot-admin:${IMAGE_VERSION} ."
            }
        }

        stage('Push Docker Image') {
            steps {
                sh "docker push ${DOCKER_REGISTRY}/iboot-admin:${IMAGE_VERSION}"
            }
        }

        stage('Deploy') {
            steps {
                sh "ssh user@server 'cd /opt/iboot && docker-compose pull && docker-compose up -d'"
            }
        }
    }

    post {
        success {
            echo 'Deployment completed successfully!'
        }
        failure {
            echo 'Deployment failed!'
        }
    }
}
```

### 8.23 备份策略

**数据库备份脚本：**

```bash
#!/bin/bash

BACKUP_DIR="/backup/mysql"
DATE=$(date +%Y%m%d_%H%M%S)
DB_NAME="iboot"
DB_USER="root"
DB_PASS="root"

# 创建备份目录
mkdir -p $BACKUP_DIR

# 备份数据库
mysqldump -u$DB_USER -p$DB_PASS $DB_NAME > $BACKUP_DIR/iboot_$DATE.sql

# 压缩备份文件
gzip $BACKUP_DIR/iboot_$DATE.sql

# 删除 30 天前的备份
find $BACKUP_DIR -name "iboot_*.sql.gz" -mtime +30 -delete

echo "Backup completed: iboot_$DATE.sql.gz"
```

## 常见问题

### 8.24 启动问题

**问题 1：端口被占用**

```
Error: Port 8080 was already in use.
```

**解决方案：**
```bash
# 查找占用端口的进程
lsof -i :8080
netstat -tlnp | grep 8080

# 杀死进程
kill -9 <PID>

# 或修改配置
server.port=8081
```

**问题 2：数据库连接失败**

```
com.mysql.cj.jdbc.exceptions.CommunicationsException: Communications link failure
```

**解决方案：**
1. 检查 MySQL 是否启动
2. 检查数据库连接配置
3. 检查防火墙设置
4. 检查数据库用户权限

### 8.25 性能问题

**问题：应用启动慢**

**解决方案：**
```bash
# 增加 JVM 堆内存
JAVA_OPTS="-Xms2g -Xmx4g"

# 启用懒加载
spring.jpa.properties.hibernate.enable_lazy_load_no_trans=true

# 优化数据库连接池
spring.datasource.hikari.maximum-pool-size=20
spring.datasource.hikari.minimum-idle=5
```

### 8.26 Docker 问题

**问题：容器无法启动**

```
Error starting container: address already in use
```

**解决方案：**
```bash
# 查看端口占用
docker-compose ps
netstat -tlnp

# 停止冲突的服务
docker-compose down

# 修改端口映射
# 编辑 docker-compose.yml，将端口改为未被占用的端口
```
